def swapcase(str)
  word_array = str.split (' ') 
  word_array.map! do |word|
    new_str = word.split(//).map! do |chr|
      if chr.match?(/[^a-zA-Z]/)
        chr
      elsif chr == chr.upcase
        chr.downcase
      else
        chr.upcase
      end
    end
    new_str.join
  end
  word_array.join(" ")
end

def swapcase(string)
  characters = string.chars.map do |char|
    if char =~ /[a-z]/
      char.upcase
    elsif char =~ /[A-Z]/
      char.downcase
    else
      char
    end
  end
  characters.join
end
